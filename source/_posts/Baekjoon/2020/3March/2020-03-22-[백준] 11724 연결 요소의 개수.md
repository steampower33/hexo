## 2. 코드

- c++

```c++
#include <iostream>
#include <vector>
#include <queue>
#include <cstring>
using namespace std;

vector<int> graph[1001];
bool visited[1001];

void bfs(int start);
void dfs(int v);

void bfs(int start){
    queue<int> q;

    q.push(start);

    while (!q.empty()){
        int node = q.front();
        q.pop();

        if(!visited[node]){
            for(int i = 0; i < graph[node].size(); i++){
                q.push(graph[node][i]);
            }
            visited[node] = true;
        }
    }
}

void dfs(int v){
    visited[v] = true;

    for(int i = 0; i < graph[v].size(); i++){
        if (!visited[graph[v][i]]){
            dfs(graph[v][i]);
        }
    }
}

int main(){
    int n = 0, m = 0, a = 0, b = 0, cnt = 0;
    scanf("%d %d", &n, &m);

    for(int i = 0; i < m; i++){
        scanf("%d %d", &a, &b);
        graph[a].push_back(b);
        graph[b].push_back(a);
    }

    memset(visited, false, sizeof(visited));
    for(int i = 1; i <= n; i++){
        if (!visited[i]){
            cnt++;
            bfs(i);
            // dfs(i);
        }
    }

    printf("%d", cnt);
    return 0;
}

```
