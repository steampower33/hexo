## 2. 코드

- c++

```c++
#include <iostream>
using namespace std;

int main() {
    int n = 0, sum = 0;
    cin >> n;
    int dp[1001][10] = { 0,};

    for(int i = 0; i < 10; i++){
        dp[1][i] = 1;
    }

    for(int i = 2; i <= 1000; i++){
        for(int j = 9; j >= 0; j--){
            if( j == 9 ) {
                dp[i][j] = 1;
            }
            else if( j == 8) {
                dp[i][j] = i;
            }
            else{
                dp[i][j] = (dp[i - 1][j] + dp[i][j + 1]) % 10007;
            }
        } 
    }

    for(int i = 0; i < 10; i++){
        sum = (sum + dp[n][i]) % 10007;
    }

    cout << sum << "\n";


    return 0;
}
```

- python

```python
import sys
input = sys.stdin.readline

if __name__ == "__main__":
    n = int(input())
    line_sum = 0
    value = 0
    dp = [[0 for i in range(10)] for j in range(1001)]
    for i in range(10): dp[1][i] = 1

    for i in range(2, 1001):
        for j in range(10):
            if j == 0:
                for k in range(10): line_sum += dp[i-1][k]
                dp[i][j] = line_sum
                line_sum = 0
            else:
                dp[i][j] = dp[i][j - 1] - dp[i - 1][j - 1]

    for i in range(10):
        value += dp[n][i]
        value = value % 10007

    print(value)
```
