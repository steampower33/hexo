## 2. 코드

```c++
#include <iostream>
#include <cstdio>
using namespace std;

int main(){
    int a, b;
    scanf("%d %d", &a, &b);

    printf("%d\n", a + b);
    printf("%d\n", a - b);
    printf("%d\n", a * b);
    printf("%d\n", a / b);
    printf("%d\n", a % b);
}
```
